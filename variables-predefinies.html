code/get2.php

 1 <?php
 2 // ce document html contient un formulaire avec un champ nommé `titre` dont les données sont envoyées par la méthode `get`
 3 
 4 // on définit des valeurs par défaut que l'utilisateur verra avant d'avoir envoyé le formulaire
 5 $titre = "";
 6 $message = "merci de remplir le formulaire et de le valider";
 7 
 8 // vérifions que :
 9 // - le couple clé / valeur `$_GET['titre']` existe bien avec la fonction `isset()`
10 // - la valeur `$_GET['titre']` n'est pas vide avec la fonction `empty()` et l'opérateur de négation `!`
11 if (isset($_GET['titre']) && !empty($_GET['titre'])) {
12     // l'utilisateur a bien envoyé une valeur et celle-ci est non vide
13 
14     // affectation de la valeur de `$_GET['titre']` à la variable `$titre`
15     $titre = $_GET['titre'];
16 
17     // affectation d'un nouveau message pour l'utilsateur
18     $message = "merci d'avoir rempli le formulaire";
19 }
20 ?><!DOCTYPE html>
21 <html>
22 <head>
23 <meta charset="utf-8" />
24 <title></title>
25 </head>
26 <body>
27 
28 <p><?php echo $message ?></p>
29 
30 <form action="" method="get">
31     <label for="titre">titre</label><br />
32     <input name="titre" type="text" value="<?php echo $titre; ?>" /><br />
33     <input type="submit" value="envoyer" />
34 </form>
35 
36 </body>
37 </html>

code/server1.php

 1 <!DOCTYPE html>
 2 <html>
 3 <head>
 4 <meta charset="utf-8" />
 5 <title></title>
 6 </head>
 7 <body>
 8 
 9 <?php
10 // la variable `$_SERVER` est un tableau
11 // elle contient des données sur le serveur, le script en cours d'exécution et le navigateur qui a lancé la requête
12 // on boucle sur tous les couples clé / valeur de la variable `$_SERVER`
13 foreach ($_SERVER as $cle => $valeur) {
14     // affichage d'un couple clé / valeur
15     echo $cle . ' = ' . $valeur;
16 
17     // affichage d'un saut de ligne html
18     // note : br = break = cassure
19     echo '<br />';
20 }
21 ?>
22 
23 </body>
24 </html>

code/session1-get.php

 1 <?php
 2 // démarrage ou récupération d'une session en cours pour le navigateur qui a fait la requête
 3 session_start();
 4 ?><!DOCTYPE html>
 5 <html>
 6 <head>
 7 <meta charset="utf-8" />
 8 <title></title>
 9 </head>
10 <body>
11 
12 <?php
13 // la variable `$_SESSION` est un tableau
14 // elle contient des données stockées dans la session en cours
15 // on boucle sur tous les couples clé / valeur de la variable `$_SESSION`
16 foreach ($_SESSION as $cle => $valeur) {
17     // affichage d'un couple clé / valeur et du type de donnée de la valeur
18     echo $cle . ' = (' . gettype($valeur) . ') ' . $valeur;
19 
20     // affichage d'un saut de ligne html
21     // note : br = break = cassure
22     echo '<br />';
23 }
24 ?>
25 
26 </body>
27 </html>
