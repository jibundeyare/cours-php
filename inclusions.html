code/include.php

1 <?php
2 // inclusion d'un autre fichier dans le fichier actuel
3 include 'hello.php';

code/include-warning.php

1 <?php
2 // inclusion d'un fichier qui n'existe pas
3 // cette inclusion génère un warning : "PHP Warning:  include(foo.php): failed to open stream: No such file or directory"
4 include 'foo.php';
5 
6 // le script continue son exécution
7 echo "ce message s'affiche";
8 echo PHP_EOL;

code/require-fatal-error.php

1 <?php
2 // inclusion d'un fichier qui n'existe pas
3 // cette inclusion génère une erreur : "PHP Warning:  require(foo.php): failed to open stream: No such file or directory"
4 require 'foo.php';
5 
6 // le script s'est arrêté
7 echo "ce message ne s'affiche pas";
8 echo PHP_EOL;

code/user-input.php

1 <?php
2 // inclusion d'une librairie pour ajouter des fonctionnalités
3 require 'lib-user-input.php';
4 
5 // utilisation des fonctions de cette librairie
6 $user_input = user_input_get_answer('entrez quelque chose :');
7 
8 echo "vous avez entré '$user_input'";
9 echo PHP_EOL;

code/include_multi.php

1 <?php
2 // inclusion 10 fois du même fichier
3 for ($i = 0; $i < 10; $i++) {
4     include 'hello.php';
5 }

code/include_once1.php

1 <?php
2 // inclusion 10 fois du même fichier
3 for ($i = 0; $i < 10; $i++) {
4     // avec le mot clé `include_once`, php s'assure que le fichier n'est inclus qu'une seule fois
5     include_once 'hello.php';
6 }

code/include_multi_fatal_error.php

1 <?php
2 // inclusion 10 fois du même fichier
3 for ($i = 0; $i < 10; $i++) {
4     // au 2ème tour, cette inclusion génère une erreur : "PHP Fatal error:  Cannot redeclare user_input_get_answer()"
5     include 'lib-user-input.php';
6 }
