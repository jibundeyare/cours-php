Ce script interroge le serveur de test de la société Paybox avec un numéro de carte bleu de test.

Exemple d’implémentation - Paybox

code/paybox.php

 1 <?php
 2 // code adapté depuis la source:
 3 //  Exemple d'implémentation - Paybox
 4 //  http://www1.paybox.com/espace-integrateur-documentation/les-solutions-paybox-direct-et-paybox-direct-plus/exemple-dimplementation/
 5 ?><!DOCTYPE html>
 6 <html>
 7 <head>
 8 <meta charset="utf-8">
 9 <title>Test Paybox direct</title>
10 </head>
11 <body>
12 <h1>Test Paybox direct</h1>
13 <?php
14 
15 session_start();
16 
17 if (!isset($_SESSION['NUMQUESTION'])) {
18     // NUMQUESTION doit être compris entre 1 et 2147483647
19     $_SESSION['NUMQUESTION'] = '1000000000';
20 } else {
21     $_SESSION['NUMQUESTION']++;
22 }
23 
24 // initialisation de la session https
25 $curl = curl_init('https://preprod-ppps.paybox.com/PPPS.php');
26 
27 // Précise que la réponse est souhaitée
28 curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
29 
30 // Présise que le session est nouvelle
31 curl_setopt($curl, CURLOPT_COOKIESESSION, true);
32 
33 $postfields = array(
34     'VERSION'       => '00104',
35     'TYPE'          => '00001',
36     'SITE'          => '1999888',
37     'RANG'          => '32',
38     'IDENTIFIANT'   => '107904482',
39     'CLE'           => '1999888I',
40 
41     'NUMQUESTION'   => $_SESSION['NUMQUESTION'],
42     'MONTANT'       => '1000',
43     'DEVISE'        => '978',
44     'REFERENCE'     => 'Hello World',
45 
46     'PORTEUR'       => '1111222233334444',
47     'DATEVAL'       => '1215',
48     'CVV'           => '123',
49 
50     'DATEQ'         => date('dmY'),
51 
52     // 'ERRORCODETEST' => '14', // Numéro de porteur invalide
53     // 'ERRORCODETEST' => '17', // Annulation client
54     // 'ERRORCODETEST' => '51', // Provision insuffisante ou crédit dépassé
55     // 'ERRORCODETEST' => '54', // Date de validité de la carte dépassée
56 );
57 
58 // Crée la chaine url encodée selon la RFC1738 à partir du tableau de paramètres séparés par le caractère &
59 $trame = http_build_query($postfields, '', '&');
60 
61 // Présise le type de requête HTTP : POST
62 curl_setopt($curl, CURLOPT_POST, true);
63 
64 // Présise le Content-Type
65 curl_setopt($curl,CURLOPT_HTTPHEADER, array('Content-Type: application/x-www-form-urlencoded'));
66 
67 // Ajoute les paramètres
68 curl_setopt($curl, CURLOPT_POSTFIELDS, $trame);
69 
70 // Envoi de la requête et obtention de la réponse
71 $raw_response = curl_exec($curl);
72 
73 $response = array();
74 parse_str($raw_response, $response);
75 
76 echo "<pre>";
77 echo "Réponse Paybox direct pour la demande 'autorize' ";
78 var_dump($raw_response);
79 var_dump($response);
80 if ($response['CODEREPONSE'] == '00000') {
81     echo 'Demande traitée avec succès';
82 } else {
83     echo 'Échec de la demande';
84 }
85 echo "</pre>";
86 
87 // fermeture de la session
88 curl_close($curl);
89 
90 ?>
91 </body>
92 </html>
